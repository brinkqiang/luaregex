cmake_minimum_required(VERSION 3.2)

PROJECT(luaregex)
INCLUDE(cmake/ModuleImport.cmake)
INCLUDE(cmake/LuaModule.cmake)
LIST(APPEND CMAKE_MODULE_PATH ${CMAKE_CURRENT_SOURCE_DIR}/cmake)
INCLUDE(cmake/ModuleCompileOptions.cmake)
ModuleSetCompileOptions()
ModuleSetWinCompilerFlags()

SET(CMAKE_CXX_STANDARD 17)
SET(CMAKE_CXX_STANDARD_REQUIRED ON)

SET(LUA_MODULE lua54)
SET(LUAREGEX_VERSION_MAJOR "1")
SET(LUAREGEX_VERSION_MINOR "0")
SET(LUAREGEX_VERSION_PATCH "1")
SET(LUAREGEX_VERSION "${LUAREGEX_VERSION_MAJOR}.${LUAREGEX_VERSION_MINOR}.${LUAREGEX_VERSION_PATCH}")

MESSAGE(STATUS "VERSION: ${LUAREGEX_VERSION}")

ADD_DEFINITIONS(-DUSE_LUAREGEX)
ADD_DEFINITIONS(-DLUA_COMPAT_MODULE -DLUA_COMPAT_APIINTCASTS)

LuaModuleImport("${LUA_MODULE}" "luaregex" "src/luaregex" "")

IF (WIN32)
ADD_CUSTOM_COMMAND(
    TARGET luaregex
    PRE_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
    ${CMAKE_CURRENT_SOURCE_DIR}/config/lua_dkm_debug.json
    $<TARGET_FILE_DIR:luaregex>
)
ENDIF(WIN32)
